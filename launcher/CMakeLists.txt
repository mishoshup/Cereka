cmake_minimum_required(VERSION 3.16)
project(Cereka LANGUAGES CXX)

set(CEREKA_QT_VERSION "6.8.3" CACHE STRING "Qt version installed via aqtinstall")
if(WIN32)
    list(APPEND CMAKE_PREFIX_PATH "C:/Qt/${CEREKA_QT_VERSION}/win64_msvc2022_64")
else()
    list(APPEND CMAKE_PREFIX_PATH "$ENV{HOME}/Qt/${CEREKA_QT_VERSION}/linux_gcc_64")
endif()

find_package(Qt6 COMPONENTS Widgets REQUIRED)

set(CMAKE_AUTOMOC ON)

set(ASSETS_DIR ${CMAKE_CURRENT_SOURCE_DIR}/template_assets)
set(EMBED_HEADER ${CMAKE_CURRENT_BINARY_DIR}/embedded_assets.h)

function(embed_file VAR_NAME FILE_PATH OUT_VAR)
    file(READ ${FILE_PATH} hex HEX)
    string(LENGTH "${hex}" hex_len)
    math(EXPR byte_len "${hex_len} / 2")
    string(REGEX REPLACE "([0-9a-fA-F][0-9a-fA-F])" "0x\\1, " arr "${hex}")
    set(${OUT_VAR}
        "static const unsigned char ${VAR_NAME}[] = { ${arr}};
static const unsigned int  ${VAR_NAME}_len = ${byte_len}u;"
        PARENT_SCOPE)
endfunction()

embed_file(kBgPng        ${ASSETS_DIR}/placeholder_bg.png      EMBED_BG)
embed_file(kCharPng      ${ASSETS_DIR}/placeholder_char.png    EMBED_CHAR)
embed_file(kSfxWav       ${ASSETS_DIR}/placeholder_sfx.wav     EMBED_SFX)
embed_file(kBgmWav       ${ASSETS_DIR}/placeholder_bgm.wav     EMBED_BGM)
embed_file(kMontserratTtf ${ASSETS_DIR}/Montserrat-Medium.ttf  EMBED_FONT)

file(WRITE ${EMBED_HEADER}
"// Auto-generated by launcher/CMakeLists.txt — do not edit
#pragma once

${EMBED_BG}
${EMBED_CHAR}
${EMBED_SFX}
${EMBED_BGM}
${EMBED_FONT}
")

add_executable(CerekaLauncher
    main.cpp
    config.cpp
    project_manager.cpp
)

target_include_directories(CerekaLauncher PRIVATE
    ${CMAKE_CURRENT_BINARY_DIR}
)

target_link_libraries(CerekaLauncher PRIVATE Qt6::Widgets)

target_compile_features(CerekaLauncher PRIVATE cxx_std_17)

set_target_properties(CerekaLauncher PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}
)

file(COPY ${CMAKE_CURRENT_SOURCE_DIR}/template_assets
     DESTINATION ${CMAKE_BINARY_DIR})
